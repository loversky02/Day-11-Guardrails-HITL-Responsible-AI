# ============================================================
# Audit Log & Monitoring System
# 
# Logs all agent interactions and monitors for security threats
# ============================================================

import json
import time
from datetime import datetime
from typing import List, Dict, Any

class AuditLogger:
    """
    Comprehensive audit logging and monitoring system.
    
    What does this do?
    - Logs every request/response with metadata
    - Tracks security events (blocks, attacks, anomalies)
    - Monitors metrics and triggers alerts
    - Exports logs to JSON for analysis
    
    Why is it needed?
    - Provides forensic evidence for security incidents
    - Enables detection of attack patterns over time
    - Supports compliance and auditing requirements
    - Helps identify system weaknesses
    
    What attack does it catch that other layers don't?
    - Slow/distributed attacks that bypass rate limiting
    - Pattern-based attacks (multiple small probes)
    - Insider threats and privilege abuse
    - Zero-day attacks (unknown patterns)
    """
    
    def __init__(self):
        self.logs = []
        self.alerts = []
        
        # Monitoring thresholds
        self.thresholds = {
            "blocked_requests_per_hour": 10,
            "failed_auth_attempts": 5,
            "pii_leaks_per_hour": 3,
            "injection_attempts_per_hour": 5,
        }
        
        # Metrics
        self.metrics = {
            "total_requests": 0,
            "blocked_requests": 0,
            "injection_attempts": 0,
            "pii_redactions": 0,
            "llm_judge_failures": 0,
        }
    
    def log_event(self, event_type: str, user_id: str, 
                  input_text: str = "", output_text: str = "",
                  metadata: dict = None):
        """
        Log a security event.
        
        Args:
            event_type: Type of event (request, block, alert, etc.)
            user_id: User identifier
            input_text: User input
            output_text: Agent response
            metadata: Additional context
        """
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "event_type": event_type,
            "user_id": user_id,
            "input_text": input_text[:200],  # Truncate for storage
            "output_text": output_text[:200],
            "metadata": metadata or {},
        }
        
        self.logs.append(log_entry)
        self.metrics["total_requests"] += 1
        
        # Update specific metrics
        if event_type == "blocked":
            self.metrics["blocked_requests"] += 1
        elif event_type == "injection_detected":
            self.metrics["injection_attempts"] += 1
        elif event_type == "pii_redacted":
            self.metrics["pii_redactions"] += 1
        elif event_type == "llm_judge_fail":
            self.metrics["llm_judge_failures"] += 1
        
        # Check thresholds and trigger alerts
        self._check_thresholds()
    
    def _check_thresholds(self):
        """Check if any monitoring thresholds are exceeded."""
        # Check blocked requests
        if self.metrics["blocked_requests"] > self.thresholds["blocked_requests_per_hour"]:
            self._trigger_alert(
                "HIGH_BLOCK_RATE",
                f"Blocked requests exceeded threshold: {self.metrics['blocked_requests']}"
            )
        
        # Check injection attempts
        if self.metrics["injection_attempts"] > self.thresholds["injection_attempts_per_hour"]:
            self._trigger_alert(
                "INJECTION_ATTACK",
                f"Multiple injection attempts detected: {self.metrics['injection_attempts']}"
            )
        
        # Check PII leaks
        if self.metrics["pii_redactions"] > self.thresholds["pii_leaks_per_hour"]:
            self._trigger_alert(
                "PII_LEAK_RISK",
                f"High number of PII redactions: {self.metrics['pii_redactions']}"
            )
    
    def _trigger_alert(self, alert_type: str, message: str):
        """Trigger a security alert."""
        alert = {
            "timestamp": datetime.now().isoformat(),
            "alert_type": alert_type,
            "message": message,
            "severity": "HIGH" if "ATTACK" in alert_type else "MEDIUM"
        }
        
        # Avoid duplicate alerts
        if not any(a["alert_type"] == alert_type for a in self.alerts):
            self.alerts.append(alert)
            print(f"\n🚨 ALERT [{alert['severity']}]: {alert_type}")
            print(f"   {message}")
    
    def export_logs(self, filename: str = "audit_log.json"):
        """Export logs to JSON file."""
        export_data = {
            "export_time": datetime.now().isoformat(),
            "total_entries": len(self.logs),
            "metrics": self.metrics,
            "alerts": self.alerts,
            "logs": self.logs
        }
        
        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(export_data, f, indent=2, ensure_ascii=False)
        
        print(f"\n✓ Audit log exported to {filename}")
        print(f"  Total entries: {len(self.logs)}")
        print(f"  Alerts triggered: {len(self.alerts)}")
    
    def get_summary(self) -> dict:
        """Get summary statistics."""
        return {
            "total_logs": len(self.logs),
            "metrics": self.metrics,
            "alerts": len(self.alerts),
            "block_rate": self.metrics["blocked_requests"] / max(1, self.metrics["total_requests"])
        }
